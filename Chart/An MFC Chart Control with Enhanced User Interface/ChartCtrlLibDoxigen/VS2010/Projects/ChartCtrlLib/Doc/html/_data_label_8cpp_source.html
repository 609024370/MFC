153 CRect CDataWnd::CalcDataWndRect(CRect& labStrRect, CWnd*pParent, CPoint origPnt)
154 {
155  CRect wndRect = labStrRect;
156  int dataRectWidth = wndRect.Width();
157 
158  CRect parentRect;
159  pParent->GetClientRect(&parentRect);
160 
161  CRect intersectRect;
162  if (origPnt.x < parentRect.Width()/2)
163  {
164  wndRect.left = origPnt.x + LABEL_OFFS;
165  wndRect.right = wndRect.left + dataRectWidth;
166  intersectRect.IntersectRect(&parentRect, &wndRect);
167  if (!intersectRect.EqualRect(wndRect))
168  {
169  wndRect.right = parentRect.right - 1;
170  wndRect.left = wndRect.right - dataRectWidth;
171  }
172  }
173  else
174  {
175  wndRect.right = origPnt.x - LABEL_OFFS;
176  wndRect.left = wndRect.right - dataRectWidth;
177  intersectRect.IntersectRect(&parentRect, &wndRect);
178  if (!intersectRect.EqualRect(&wndRect))
179  {
180  wndRect.left = parentRect.left + 1;
181  wndRect.right = wndRect.left + dataRectWidth;
182  }
183  }
184 
185  int offsY = parentRect.bottom - wndRect.Height() - LABEL_OFFS;
186 
187  wndRect.OffsetRect(0, offsY);
