109 size_t CChart::AppendChartData(V_CHARTDATAD& vData)
110 {
111  if (vData.empty())
112  return 0;
113 
114 // Find first element to append
115  V_CHARTDATAD::iterator itStart = vData.begin();
116  if (!m_vDataPnts.empty())
117  {
118  PointD oldEndPntD = m_vDataPnts.back();
119  double oldEndX = oldEndPntD.X;
120  itStart = find_if(itStart, vData.end(), greater_or_equal<double, false>(oldEndX));
121  if (itStart != vData.end()) // Find the first not a dublicate
122  {
123  while ((itStart != vData.end())&&(oldEndPntD.X == itStart->X))
124  {
125  if (oldEndPntD.Y != itStart->Y)
126  break;
127  ++itStart;
128  }
129  }
130  if (itStart == vData.end())
131  return 0; // All data points are duplicates
132  }
133 
134 // Resize and copy
135  size_t oldSize = m_vDataPnts.size();
136  size_t sizeNew = oldSize + distance(itStart, vData.end());
137  m_vDataPnts.resize(sizeNew);
138  copy(itStart, vData.end(), m_vDataPnts.begin() + oldSize);
139 
140 // Correct minmaxes
141  m_fMinValX = m_vDataPnts.front().X; // Just in case; it was not changed
142  m_fMaxValX = m_vDataPnts.back().X;
143  std::pair<V_CHARTDATAD::iterator, V_CHARTDATAD::iterator> pair_minmaxY =
144  minmax_element(m_vDataPnts.begin(), m_vDataPnts.end(), less_pnt<double, true>());
145  m_fMinValY = pair_minmaxY.first->Y;
146  m_fMaxValY = pair_minmaxY.second->Y;
147  return sizeNew - oldSize;
148 }

150 bool CChart::TruncateChartData(double startX, double endX)
151 {
152  if (startX >= endX) // Wrong parameters
153  return false;
154  if (!HasData())
155  return false;
156  if ((m_fMinValX >= startX) && (m_fMaxValX <= endX))
157  return false; // Nothing to truncate
158 
159  PAIR_ITS pair_its = GetStartEndDataIterators(m_vDataPnts, startX, endX);
160 // We need this last point to draw the right end of the chart
161  if (pair_its.second != m_vDataPnts.end())
162  ++pair_its.second;
163 
164  size_t dataSize = distance(pair_its.first, pair_its.second);
165  if (dataSize == 0)
166  {
167  m_vDataPnts.clear();
168  m_fMinValX = DBL_MAX;
169  m_fMaxValX = -DBL_MAX;
170  m_fMinValY = DBL_MAX;
171  m_fMaxValY = -DBL_MAX;
172  }
173  else
174  {
175  V_CHARTDATAD vDataD(dataSize);
176  copy(pair_its.first, pair_its.second, vDataD.begin());
177 
178  m_vDataPnts.swap(vDataD);
179 
180  m_fMinValX = m_vDataPnts.front().X;
181  m_fMaxValX = m_vDataPnts.back().X;
182  std::pair<V_CHARTDATAD::iterator, V_CHARTDATAD::iterator> pair_minmaxY =
183  minmax_element(m_vDataPnts.begin(), m_vDataPnts.end(), less_pnt<double, true>());
184  m_fMinValY = pair_minmaxY.first->Y;
185  m_fMaxValY = pair_minmaxY.second->Y;
186  }
187  return true;

190 PAIR_ITNEAREST CChart::GetNearestPointD(const PointD& origPntD, double dist, PointD& selPntD)
191 {
192  V_CHARTDATAD::iterator it = m_vDataPnts.begin(), itE = m_vDataPnts.end();
193  int nmbMultPntsD = 0;
194  double leftX = origPntD.X - dist/2.0;
195  double rightX = origPntD.X + dist/2.0;
196 // Find the first point in distance range from the origPntsD.X, if it exists
197  it = find_if(it, itE, coord_in_range<double, false>(leftX, rightX));
198  if (it != itE) // Find closest to origPntD.X
199  {
200  it = find_nearest(it, itE, nearest_to<double, false>(origPntD, dist));
201  if (it != itE) // Impossible; will return found_if at least
202  {
203  selPntD = *it; // Get number of multivalued points (the same X's, different Y's)
204  nmbMultPntsD = count_if(m_vDataPnts.begin(), it, count_in_range<double, false>(selPntD.X, selPntD.X));
205  return make_pair(it, nmbMultPntsD + 1);
206  }
207  }
208  return make_pair(itE, 0);
209 }

293

302  if (vSize == 1)

306  }
307  }
308 
309  if (grPathPtr->GetPointCount() > 0) // Has points to draw
310  {
311  pen.SetWidth(1.0f*dpiRatio);
312  pen.SetDashStyle(DashStyleSolid);
313  grPtr->DrawPath(&pen, grPathPtr);
314  if (((m_dashStyle == DashStyleCustom)||(vSize == 1))&& m_bSelected && (dpiRatio == 1.0f))
315  {
316  pen.SetColor(Color(SetAlpha(m_colChart, ALPHA_SELECT)));
317  pen.SetWidth(m_fPenWidth + PEN_SELWIDTH);

325 bool CChart::ConvertChartData(V_CHARTDATAD& vDataPnts, V_CHARTDATAF& vDataPntsF,
326  MatrixD* pMatrixD, double startX, double endX)
327 {
328  ENSURE(vDataPnts.size() > 0);
329  V_CHARTDATAD::iterator itFirst, itLast;
330  if (vDataPnts.size() == 1)
331  {
332  itFirst = vDataPnts.begin();
333  itLast = vDataPnts.end();
334  }
335  else
336  std::tie(itFirst, itLast) = GetStartEndDataIterators(vDataPnts, startX, endX);
337  if (itLast != vDataPnts.end()) // Last iterator points after the last point
338  ++itLast;
339 
340  int vSize = distance(itFirst, itLast);
341  if (vSize == 0)
342  return false;

344  vDataPntsF.resize(vSize);
345  transform(itFirst, itLast, vDataPntsF.begin(), TRANSFORM_TO_PNTF(m_fLocScaleY, pMatrixD));
346
