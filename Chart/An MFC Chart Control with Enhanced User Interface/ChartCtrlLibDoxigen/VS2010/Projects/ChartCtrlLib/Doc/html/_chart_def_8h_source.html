65  { m_offsX += offsX*m_scX; m_offsY += offsY*m_scY; }

68 // Operations on matrixD; if the matrix is not invertible, returns false
69  bool Invert(void)
70  {
71  if (m_scX*m_scY == 0.0)
72  return false;
73 
74  m_offsX = -m_offsX/m_scX;
75  m_offsY = -m_offsY/m_scY;
76  m_scX = 1.0/m_scX;
77  m_scY = 1.0/m_scY;
78  return true;
79  }

85  pMatrix->m_scY = m_scY;

100  PointD TransformToPntD(double locScY, const Gdiplus::PointF& pntF)
101  {
102  ENSURE(m_scX*m_scY != 0.0);
103  MatrixD* matrixDI = Clone();
104  matrixDI->Invert();
105 
106  double X = pntF.X*matrixDI->m_scX + matrixDI->m_offsX;
107  double Y = pntF.Y*matrixDI->m_scY + matrixDI->m_offsY;
108  Y /= locScY;
109 
110  delete matrixDI;
111  return PointD(X, Y);
112  }
